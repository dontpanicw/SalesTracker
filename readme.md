```sql
    SELECT 
        amount,
        PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY amount) OVER () as median,
        PERCENTILE_CONT(0.9) WITHIN GROUP (ORDER BY amount) OVER () as percentile_90
    FROM items
    WHERE date >= $1 AND date <= $2
)

    COALESCE(SUM(amount), 0) as sum,
    COALESCE(AVG(amount), 0) as avg,
    COUNT(*) as count,
    COALESCE(MAX(median), 0) as median,
    COALESCE(MAX(percentile_90), 0) as percentile_90
FROM stats
```
